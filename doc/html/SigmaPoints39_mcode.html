1%% SigmaPoints -- sigma point sampling function
 2% Inputs -----------------------------
 3%     x    : state vector mean
 4%     P    : state vector covariance
 5%     c    : UKF paramater
 6% Outputs ----------------------------
 7%     X : sigma points matrix
 8%%
 9function X = SigmaPoints(X,x,P,c)
10%#codegen
11
12coder.inline('never')
13
14A = c*chol(P)';
15Y = x(:,ones(1,numel(x)));
16X = [x Y+A Y-A];
